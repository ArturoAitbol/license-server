                customer.subaccountId = subaccountDetails.id;
                const subaccountLicenses = licences.filter((l: License) => (l.subaccountId === subaccountDetails.id));
                if (subaccountLicenses.length > 0) {

                } else {
                    customer.status = 'Inactive';

        });
